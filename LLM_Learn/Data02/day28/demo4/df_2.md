# 生成时间序列


```python
import pandas as pd
# 指定了起始和结束 生成序列
print(pd.date_range("2022-10-10", "2022-10-20"))
```

    DatetimeIndex(['2022-10-10', '2022-10-11', '2022-10-12', '2022-10-13',
                   '2022-10-14', '2022-10-15', '2022-10-16', '2022-10-17',
                   '2022-10-18', '2022-10-19', '2022-10-20'],
                  dtype='datetime64[us]', freq='D')
    


```python
# 只指定起始
print(pd.date_range("2022-12-12", periods=5))
```

    DatetimeIndex(['2022-12-12', '2022-12-13', '2022-12-14', '2022-12-15',
                   '2022-12-16'],
                  dtype='datetime64[us]', freq='D')
    


```python
# 指定起始时间 以及 指定频率

print(pd.date_range("2022-12-12", periods=5, freq="h"))

```

    DatetimeIndex(['2022-12-12 00:00:00', '2022-12-12 01:00:00',
                   '2022-12-12 02:00:00', '2022-12-12 03:00:00',
                   '2022-12-12 04:00:00'],
                  dtype='datetime64[us]', freq='h')
    


```python
# 生成季度时间序列
# 默认情况 3 6 9 12
# 季度频率 ： 1 4 7 10
print(pd.date_range("2015-05-03", periods=10, freq="QE-JAN"))


```

    DatetimeIndex(['2015-07-31', '2015-10-31', '2016-01-31', '2016-04-30',
                   '2016-07-31', '2016-10-31', '2017-01-31', '2017-04-30',
                   '2017-07-31', '2017-10-31'],
                  dtype='datetime64[us]', freq='QE-JAN')
    


```python
# 指定具体到周几的频率

print(pd.date_range("2026-05-13", periods=10, freq="W-FRI"))

```

    DatetimeIndex(['2026-05-15', '2026-05-22', '2026-05-29', '2026-06-05',
                   '2026-06-12', '2026-06-19', '2026-06-26', '2026-07-03',
                   '2026-07-10', '2026-07-17'],
                  dtype='datetime64[us]', freq='W-FRI')
    


```python
# 指定小时和分钟 生成时间序列
print(pd.date_range("2026-05-13", periods=10, freq="2h30min"))
```

    DatetimeIndex(['2026-05-13 00:00:00', '2026-05-13 02:30:00',
                   '2026-05-13 05:00:00', '2026-05-13 07:30:00',
                   '2026-05-13 10:00:00', '2026-05-13 12:30:00',
                   '2026-05-13 15:00:00', '2026-05-13 17:30:00',
                   '2026-05-13 20:00:00', '2026-05-13 22:30:00'],
                  dtype='datetime64[us]', freq='150min')
    


```python
# 统计最高温度 最低温度的平均值

df = pd.read_csv("weather.csv",parse_dates=["date"])

df.set_index("date", inplace=True)
print(df[["temp_max", "temp_min"]].resample("YE").mean())
```
